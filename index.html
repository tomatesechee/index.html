    <script>

        const btnOui = document.getElementById('btnOui');

        const btnNon = document.getElementById('btnNon');

        const container = document.getElementById('container');

        const successMsg = document.getElementById('success-msg');

        let autoMoveInterval;

        btnNon.addEventListener('mouseover', fuiteBouton);

        btnNon.addEventListener('touchstart', fuiteBouton);

        setTimeout(() => {

             autoMoveInterval = setInterval(fuiteBouton, 1000);

        }, 500);

        function fuiteBouton() {

            const margin = 50;

            const windowWidth = window.innerWidth - margin;

            const windowHeight = window.innerHeight - margin;

           

            const btnWidth = btnNon.offsetWidth;

            const btnHeight = btnNon.offsetHeight;

            const randomX = Math.max(margin, Math.random() * (windowWidth - btnWidth));

            const randomY = Math.max(margin, Math.random() * (windowHeight - btnHeight));

            btnNon.style.position = 'fixed';

            btnNon.style.left = randomX + 'px';

            btnNon.style.top = randomY + 'px';

            btnNon.style.zIndex = '15';

        }

        btnOui.addEventListener('click', () => {

            clearInterval(autoMoveInterval);

            container.style.display = 'none';

            btnNon.style.display = 'none';

            successMsg.style.display = 'block';

            document.body.style.backgroundColor = '#88f2a6';

            document.body.style.backgroundImage = 'none';

        });
